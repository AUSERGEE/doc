# Function.apply()

指定this

  var numbers = [5, 6, 2, 3, 7];
  var max = Math.max.apply(null, numbers);
  console.log(max);  //7

  var array = ['a', 'b'];
  var elements = [0, 1, 2];
  array.push.apply(array, elements);
  console.info(array); // ["a", "b", 0, 1, 2]
