# for..in

遍历对象key值：

const obj = { id:1,name:'zhangsan',age:18}
 
for(let key  in obj){
    console.log(key + '---' + obj[key])
}

# Object.keys（obj）

console.log(Object.keys(obj))  //['id','name','age'] console.log(Object.values(obj))  //['1','zhangsan','18']
